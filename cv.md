### _Code example:_

Given an array of integers, you need to expand the array in reverse order.

```
let arr = [1, 2, 3, 4, 5, 6];

function revArray(arr) {
    tempArray = [];
    for (let i = arr.length - 1; i >= 0; i--) {
        tempArray.push(arr[i]);
    }
    return tempArray;
}
console.log(revArray(arr));
```
